```tsx

 * hideActions:
 *  - CSB
 *  - EXTERNAL
 */

const style = { style: { marginLeft: "10px" } };
```

```tsx
   <React.Fragment>
```

## block 按钮

```tsx

 * hideActions:
 *  - CSB
 *  - EXTERNAL

    </Button>

  </React.Fragment>
```
